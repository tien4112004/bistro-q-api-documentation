60 if (order == null)

64
65 order.OrderItems = _mapper.Map<List<OrderItem>>(order.OrderItems);

69#region braindead-code

124 var updatedOrder = _mapper.Map<Order>(order);

126 List<OrderItem> addedItems = new List<OrderItem>();

160 public async Task<IEnumerable<OrderItemDto>> CancelOrderItems(int tableId, [FromBody] IEnumerable<RemoveOrderItemRequestDto> orderItems)
161 {
162 var order = await _unitOfWork.OrderRepository.GetByTableIdAsync(tableId);
163 if (order == null)
164 {
165 throw new ResourceNotFoundException("Order not found");
166 }
167
168 var updatedOrder = order;
169
170 List<OrderItem> removedItems = new List<OrderItem>();
171 foreach (var item in orderItems)
172 {
173 var orderItem = order.OrderItems.FirstOrDefault(oi => oi.OrderItemId == item.OrderItemId);
174 if (orderItem == null)
175 {
176 throw new ResourceNotFoundException("OrderItem not found");
177 }
178
179 await _unitOfWork.OrderItemRepository.DeleteAsync(orderItem);
180 removedItems.Add(orderItem);
181 updatedOrder.TotalAmount -= orderItem.Quantity * orderItem.PriceAtPurchase;
182 }
183 await _unitOfWork.OrderRepository.UpdateAsync(order, updatedOrder);
184 await _unitOfWork.SaveChangesAsync();
