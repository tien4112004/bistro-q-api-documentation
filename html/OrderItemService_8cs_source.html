44 public async Task<IEnumerable<OrderItemDto>> UpdateOrderItemsStatusAsync(IEnumerable<string> orderItems, OrderItemStatus status)
45 {
46 var enumerable = orderItems as string[] ?? orderItems.ToArray();
47
48 var updatedOrderItems = new List<OrderItem>();
49
50 decimal totalCancelAmount = 0;
51 await _unitOfWork.BeginTransactionAsync();
52 try
53 {
54 for (var i = 0; i < enumerable.Count(); i++)
55 {
56 var orderItem = await _unitOfWork.OrderItemRepository.GetByIdAsync(enumerable[i]);
57 if (orderItem == null)
58 {
59 throw new ResourceNotFoundException($"Order item with id {enumerable[i]} not found.");
60 }
61
62 OrderItemStatusTransition.UpdateStatus(orderItem, status);
63 totalCancelAmount += orderItem.Quantity.Value * orderItem.PriceAtPurchase.Value;
64 updatedOrderItems.Add(orderItem);
65 }
66
67 var order = await _unitOfWork.OrderRepository.GetByIdAsync(updatedOrderItems.First().OrderId);
68
69 if (status == OrderItemStatus.Cancelled)
70 {
71 order.TotalAmount -= totalCancelAmount;
72 }
73 await _unitOfWork.OrderRepository.UpdateAsync(order, order);
74
75 await _unitOfWork.SaveChangesAsync();
76 await _unitOfWork.CommitTransactionAsync();
77 }
78 catch (Exception e)
79 {
80 await _unitOfWork.RollbackTransactionAsync();
81 throw;
82 }
83
84 return _mapper.Map<IEnumerable<OrderItemDto>>(updatedOrderItems);
85 }
