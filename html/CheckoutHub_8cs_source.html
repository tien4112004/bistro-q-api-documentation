31 public async Task InitiateCheckout(int tableId)
32 {
33 try
34 {
35 var order = await _orderService.GetOrderByTableId(tableId);
36 if (order == null)
37 {
38 throw new ResourceNotFoundException("Order not found");
39 }
40
41 await _orderService.UpdateStatus(tableId, OrderStatus.Pending);
42
43 var table = await _tableService.GetByIdAsync(tableId);
44 var paymentData = await _paymentService.InitiatePayment(_mapper.Map<OrderDto>(order));
45
46 Console.WriteLine($"Payment initiated for table {tableId} in zone {table.ZoneId}");
47
48 await Clients.Caller.SendAsync("CheckoutInitiated", paymentData);
49 await Clients.Group("Cashiers").SendAsync("NewCheckout", tableId, table.Number, table.ZoneName);
50 }
51 catch (Exception e)
52 {
53 Console.WriteLine(e.StackTrace);
54 await Clients.Caller.SendAsync("CheckoutError", e.Message);
55 }
56 }
57
58 public async Task CompleteCheckout(int tableId)
59 {
60 try
61 {
62 var order = await _orderService.GetOrderByTableId(tableId);
63 if (order == null)
64 {
65 throw new ResourceNotFoundException("Order not found");
66 }
67
68 Console.WriteLine($"Payment completed for table {tableId}");
69
70 await _orderService.UpdateStatus(tableId, OrderStatus.Completed);
71 await Clients.Group($"Client_{tableId}").SendAsync("CheckoutCompleted");
72 }
73 catch (Exception e)
74 {
75 Console.WriteLine(e.StackTrace);
76 await Clients.Caller.SendAsync("CheckoutError", e.Message);
77 }
78 }

80 public override async Task OnConnectedAsync()
81 {
82 var userRole = Context.User.Claims.FirstOrDefault(c => c.Type == ClaimTypes.Role)?.Value;
83 if (userRole == BistroRoles.Cashier)
84 {
85 await Groups.AddToGroupAsync(Context.ConnectionId, "Cashiers");
86 }
87 else if (userRole == BistroRoles.Client)
88 {
89 var user = await _userManager.GetUserAsync(Context.User);
90
91 await Groups.AddToGroupAsync(Context.ConnectionId, "Client_" + user.TableId.Value);
92 }
93 }
