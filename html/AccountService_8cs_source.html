49public async Task<(IEnumerable<AccountDto> Accounts, int Count)> GetAllAsync(
50 AccountCollectionQueryParams query)
51{
52 var usersWithRoles = _context.Users
53 .Include(u => u.Table)
54 .ThenInclude(t => t.Zone)
55 .Select(u => new
56 {
57 User = u,
58 Role = _context.UserRoles
59 .Where(ur => ur.UserId == u.Id)
60 .Join(_context.Roles,
61 ur => ur.RoleId,
62 r => r.Id,
63 (ur, r) => r.Name)
64 .FirstOrDefault()
65 });
66
67 if (!string.IsNullOrWhiteSpace(query.Username))
68 {
69 usersWithRoles = usersWithRoles.Where(x => EF.Functions.Like(
70 x.User.UserName,
71 $"%{query.Username}%"
72 ));
73 }
74
75 if (!string.IsNullOrWhiteSpace(query.Role))
76 {
77 usersWithRoles = usersWithRoles.Where(u => u.Role == query.Role);
78 }
79
80 if (!string.IsNullOrWhiteSpace(query.Zone))
81 {
82 usersWithRoles = usersWithRoles.Where(u => u.User.Table != null
83 && u.User.Table.Zone != null
84 && u.User.Table.Zone.Name == query.Zone);
85 }
86
87 var totalCount = await usersWithRoles.CountAsync();
88
89 var sortedUsers = query.OrderBy?.ToLower() switch
90 {
91 "username" => query.OrderDirection == "asc"
92 ? usersWithRoles.OrderBy(u => u.User.UserName)
93 : usersWithRoles.OrderByDescending(u => u.User.UserName),
94 "role" => query.OrderDirection == "asc"
95 ? usersWithRoles.OrderBy(u => u.Role)
96 : usersWithRoles.OrderByDescending(u => u.Role),
97 _ => usersWithRoles.OrderBy(u => u.User.UserName)
98 };
99
100 var pagedUsers = await sortedUsers
101 .Skip(query.Size * (query.Page - 1))
102 .Take(query.Size)
103 .Select(u => new AccountDto
104 {
105 Id = u.User.Id,
106 Username = u.User.UserName,
107 Role = u.Role,
108 TableId = u.User.TableId,
109 Table = u.User.Table != null ? _mapper.Map<TableDetailDto>(u.User.Table) : null
110 })
111 .ToListAsync();
112
113 return (pagedUsers, totalCount);
114}

116 public async Task<AccountDto> CreateAccountAsync(CreateAccountDto request)
117 {
118 await _unitOfWork.BeginTransactionAsync();
119 try
120 {
121 var user = new AppUser
122 {
123 UserName = request.Username,
124 };
125
126 var result = await _userManager.CreateAsync(user, request.Password);
127 if (!result.Succeeded)
128 {
129 throw new Exception("Failed to create user");
130 }
131
132 var roleResult = await _userManager.AddToRoleAsync(user, request.Role);
133 if (!roleResult.Succeeded)
134 {
135 throw new Exception("Failed to assign role to user");
136 }
137
138 var createdUser = await _userManager.FindByNameAsync(user.UserName);
139 if (createdUser == null)
140 {
141 throw new ResourceNotFoundException("User not found");
142 }
143
144 if (request.TableId.HasValue)
145 {
146 var table = await _context.Tables.FindAsync(request.TableId);
147 if (table == null)
148 {
149 throw new ResourceNotFoundException("Table not found");
150 }
151
152 createdUser.TableId = table.TableId;
153 await _userManager.UpdateAsync(createdUser);
154 }
155
156 await _unitOfWork.CommitTransactionAsync();
157
158 return new AccountDto
159 {
160 Id = createdUser.Id,
161 Username = createdUser.UserName,
162 Role = request.Role
163 };
164 }
165 catch (Exception e)
166 {
167 await _unitOfWork.RollbackTransactionAsync();
168 throw;
169 }

172 public async Task<AccountDto> UpdateAccountAsync(string id, UpdateAccountDto request)
173 {
174 await _unitOfWork.BeginTransactionAsync();
175 try
176 {
177 var user = await _userManager.FindByIdAsync(id);
178 if (user == null)
179 {
180 throw new ResourceNotFoundException("User not found");
181 }
182
183 if (!string.IsNullOrEmpty(request.Username))
184 user.UserName = request.Username;
185
186 if (!string.IsNullOrEmpty(request.Password))
187 {
188 var token = await _userManager.GeneratePasswordResetTokenAsync(user);
189 var passwordResult = await _userManager.ResetPasswordAsync(user, token, request.Password);
190 if (!passwordResult.Succeeded)
191 {
192 throw new Exception("Failed to reset password");
193 }
194 }
195
196 if (!string.IsNullOrEmpty(request.Role))
197 {
198 var currentRoles = await _userManager.GetRolesAsync(user);
199 var removeResult = await _userManager.RemoveFromRolesAsync(user, currentRoles);
200 if (!removeResult.Succeeded)
201 {
202 throw new Exception("Failed to remove roles from user");
203 }
204
205 var addRoleResult = await _userManager.AddToRoleAsync(user, request.Role);
206 if (!addRoleResult.Succeeded)
207 {
208 throw new Exception("Failed to assign role to user");
209 }
210 }
211
212 if (request.TableId.HasValue)
213 {
214 var table = await _context.Tables.FindAsync(request.TableId);
215 if (table == null)
216 {
217 throw new ResourceNotFoundException("Table not found");
218 }
219
220 user.TableId = table.TableId;
221 }
222
223 var updateResult = await _userManager.UpdateAsync(user);
224 if (!updateResult.Succeeded)
225 {
226 Console.WriteLine(JsonSerializer.Serialize(updateResult.Errors));
227 throw new Exception("Failed to update user");
228 }
229
230 var updatedUser = await _userManager.FindByIdAsync(id);
231 if (updatedUser == null)
232 {
233 throw new ResourceNotFoundException("User not found");
234 }
235
236 await _unitOfWork.CommitTransactionAsync();
237
238 return new AccountDto
239 {
240 Id = updatedUser.Id,
241 Username = updatedUser.UserName,
242 Role = request.Role
243 };
244 }
245 catch (Exception e)
246 {
247 await _unitOfWork.RollbackTransactionAsync();
248 throw;
249 }
250 }
